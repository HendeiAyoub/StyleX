-- 1. Create the products table
CREATE TABLE IF NOT EXISTS public.products (
    id bigint PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
    title text NOT NULL,
    price numeric NOT NULL,
    category text NOT NULL,
    image text NOT NULL,
    description text NOT NULL,
    created_at timestamp with time zone DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 2. Enable Row Level Security (RLS)
ALTER TABLE public.products ENABLE ROW LEVEL SECURITY;

-- 3. Create a public read policy
-- This allows anyone to read the products table without requiring authorization.
DO $$
BEGIN
    IF NOT EXISTS (
        SELECT 1 FROM pg_policies WHERE tablename = 'products' AND policyname = 'Allow public read access'
    ) THEN
        CREATE POLICY "Allow public read access" 
        ON public.products 
        FOR SELECT 
        USING (true);
    END IF;
END $$;

### Step 2: Configure your API Key in Xcode

To allow the iOS app to securely retrieve items directly from this live table, you need to add your Supabase public API Key.

1. Go to your **Supabase Dashboard** -> **Project Settings** -> **API**.
2. Locate the **`anon` `public`** API Key. Click the **Copy** button.
3. In Xcode, open the file `constants.swift` located at `ShoppingApp/config/constants.swift`.
4. Replace the value of `SUPABASE_ANON_KEY` (line 3) with your copied API key:

```swift
// In constants.swift
static let SUPABASE_ANON_KEY    = "YOUR_COPIED_ANON_KEY_HERE"
```

Once both steps are complete, build and run the app. It will automatically stop using the local mock fallback and fetch these items directly from your live Supabase database! 🚀
